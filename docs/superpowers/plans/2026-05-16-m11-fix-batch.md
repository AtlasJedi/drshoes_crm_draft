```java
package com.drshoes.app.order.domain;

import java.time.Clock;
import java.time.Duration;
import java.time.Instant;
import java.util.Set;

/** Computes the "pilne" flag — >= 14 days in shop, not yet delivered/cancelled. */
public final class OrderUrgency {
    public static final int THRESHOLD_DAYS = 14;
    private static final Set<OrderStatus> EXCLUDED =
        Set.of(OrderStatus.WYDANE, OrderStatus.ANULOWANE, OrderStatus.WSTEPNIE_PRZYJETE);

    private OrderUrgency() {}

    public static boolean isUrgent(Instant receivedAt, OrderStatus status, Clock clock) {
        if (receivedAt == null) return false;
        if (EXCLUDED.contains(status)) return false;
        return Duration.between(receivedAt, clock.instant()).toDays() >= THRESHOLD_DAYS;
    }

    public static boolean isUrgent(Instant receivedAt, OrderStatus status) {
        return isUrgent(receivedAt, status, Clock.systemUTC());
    }
}
```

```java
package com.drshoes.app.messaging.util;

/** Naive HTML → plain text. Sufficient for inbound email preview/body. */
public final class HtmlStripper {

    private HtmlStripper() {}

    public static String toPlainText(String html) {
        if (html == null) return null;
        String s = html
            .replaceAll("(?is)<style[^>]*>.*?</style>", "")
            .replaceAll("(?is)<script[^>]*>.*?</script>", "")
            .replaceAll("(?is)<br\\s*/?>", "\n")
            .replaceAll("(?is)</p\\s*>", "\n")
            .replaceAll("<[^>]+>", "")
            .replace("&nbsp;", " ")
            .replace("&amp;", "&")
            .replace("&lt;", "<")
            .replace("&gt;", ">")
            .replace("&quot;", "\"")
            .replace("&#39;", "'");
        return s.replaceAll("\\s+", " ").trim();
    }
}
```

```ts
import type { OrderStatus } from "./types";

const DAY_MS = 86_400_000;
const EXCLUDED: ReadonlySet<OrderStatus> = new Set(["WYDANE", "ANULOWANE", "WSTEPNIE_PRZYJETE"]);

interface OrderTime {
  receivedAt: string | null;
  status: OrderStatus;
  pickedUpAt?: string | null;
}

export function daysInShop(o: OrderTime): number | null {
  if (!o.receivedAt) return null;
  if (EXCLUDED.has(o.status)) return null;
  const recv = new Date(o.receivedAt).getTime();
  if (!Number.isFinite(recv)) return null;
  return Math.max(0, Math.floor((Date.now() - recv) / DAY_MS));
}

export function isUrgent(o: { receivedAt: string | null; status: OrderStatus }): boolean {
  const d = daysInShop({ receivedAt: o.receivedAt, status: o.status });
  return d !== null && d >= 14;
}
```

```ts
import { test, expect, type Page } from "@playwright/test";

const BASE = process.env.E2E_BASE_URL ?? "http://localhost:3000";
const REPORT: Array<{ surface: string; defect: string | null; screenshot: string }> = [];

async function shot(page: Page, name: string) {
  const path = `e2e/_screens/m11-${name}.png`;
  await page.screenshot({ path, fullPage: true });
  return path;
}

async function loginAsAdmin(page: Page) {
  await page.goto(`${BASE}/login`);
  await page.fill('input[name="email"]', process.env.ADMIN_EMAIL ?? "admin@drshoes.pl");
  await page.fill('input[name="password"]', process.env.ADMIN_PASSWORD ?? "admin123");
  await page.click('button[type="submit"]');
  await page.waitForURL(/\/admin/);
}

test.beforeAll(async ({ browser }) => {
  const page = await browser.newPage();
  await loginAsAdmin(page);
  await page.close();
});

test("orders list — miejsce chip + urgent highlight", async ({ page }) => {
  await loginAsAdmin(page);
  await page.goto(`${BASE}/admin/orders`);
  const screen = await shot(page, "01-orders-list");
  const hasMiejsce = (await page.locator('td span:has-text("📍"), [class*="LocationChip"]').count()) > 0;
  REPORT.push({
    surface: "orders.list.miejsce",
    defect: hasMiejsce ? null : "No LocationChip visible in orders list",
    screenshot: screen
  });

test("orders list — urgent row gets magenta tint", async ({ page }) => {
  await loginAsAdmin(page);
  await page.goto(`${BASE}/admin/orders`);
  const screen = await shot(page, "02-urgent-row");
  const urgentRows = await page.locator('tr[class*="magenta"]').count();
  REPORT.push({
    surface: "orders.list.urgent",
    defect: urgentRows === 0 ? "No urgent row tinted magenta — check fixture or class" : null,
    screenshot: screen
  });
});

test("drawer — no Wykonawca + days in shop", async ({ page }) => {
  await loginAsAdmin(page);
  await page.goto(`${BASE}/admin/orders`);

  const wykonawca = await page.locator('text=Wykonawca').count();

  REPORT.push({
    surface: "drawer.wykonawca",
    defect: wykonawca > 0 ? "Wykonawca still visible in drawer" : null,
    screenshot: screen
  });
  REPORT.push({
    surface: "drawer.dni",

  });
```
